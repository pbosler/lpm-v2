295  real(kreal) :: HarmGradient(3)

297  !
298  real(kreal) :: lon, lat, u, v
299 
300  lat = latitude(xyz)
301  lon = longitude(xyz)
302 
303  u = -12.0_kreal*sqrt(35.0_kreal)*sin(4.0_kreal * lon)*(-1.0_kreal + sin(lat)*sin(lat))**2 * tan(lat)
304  v = 12.0_kreal*sqrt(35.0_kreal)*cos(4.0_kreal * lon)*cos(lat)*sin(lat)*sin(lat)*(-1.0_kreal + sin(lat)*sin(lat)) + &
