372  real(kreal), dimension(:), intent(out) :: u
373  real(kreal), dimension(:), intent(out) :: v

383  type(mpisetup), intent(in) :: mpiParticles
384  !
385  integer(kint) :: i, j, mpiErrCode
386  real(kreal) :: strength
387 
388  !$acc parallel loop gang
389  do i = mpiparticles%indexStart(procrank), mpiparticles%indexEnd(procrank)
390  u(i) = 0.0_kreal
391  v(i) = 0.0_kreal
392  w(i) = 0.0_kreal
393  !$acc loop vector
394  do j = 1, i - 1
395  if ( activemask(j) ) then
396  strength = -relvortin(j)*areain(j) / ( 4.0_kreal * pi * sphereradius * &
397  ( sphereradius * sphereradius - x(i)*x(j) - y(i)*y(j) - z(i)*z(j) ))
398  u(i) = u(i) + ( y(i)*z(j) - z(i)*y(j) ) * strength
399  v(i) = v(i) + ( z(i)*x(j) - x(i)*z(j) ) * strength
400  w(i) = w(i) + ( x(i)*y(j) - y(i)*x(j) ) * strength
401  endif
402  enddo
403  !$acc loop vector
404  do j = i + 1, size(x)
405  if ( activemask(j) ) then
406  strength = -relvortin(j)*areain(j) / ( 4.0_kreal * pi * sphereradius * &
407  ( sphereradius * sphereradius - x(i)*x(j) - y(i)*y(j) - z(i)*z(j) ))
408  u(i) = u(i) + ( y(i)*z(j) - z(i)*y(j) ) * strength
409  v(i) = v(i) + ( z(i)*x(j) - x(i)*z(j) ) * strength
410  w(i) = w(i) + ( x(i)*y(j) - y(i)*x(j) ) * strength
411  endif
412  enddo
413  enddo
