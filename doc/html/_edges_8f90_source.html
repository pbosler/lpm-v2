449 function edgeangleatorig( self, edgeIndex, aParticles )
450  real(kreal) :: edgeAngleAtOrig
451  type(edges), intent(in) :: self
452  integer(kint), intent(in) :: edgeIndex
453  type(particles), intent(in) :: aParticles
454  !
455  real(kreal) :: edge1Vec(3), newEdgeVec(3), cp(3)
456 
457  edgeangleatorig = 0.0_kreal
458  if ( aparticles%geomKind == planar_geom ) then
459  !
460  ! define angle relative to positive real axis
461  !
462  edgeangleatorig = atan2( aparticles%y( self%dest(edgeindex)) - aparticles%y( self%orig(edgeindex)), &
463  aparticles%x( self%dest(edgeindex)) - aparticles%x( self%orig(edgeindex)) )
464  elseif ( aparticles%geomKind == sphere_geom ) then
465  !
466  ! define angle relative to first edge at particle
467  !
468  if ( aparticles%nEdges( self%orig(edgeindex) ) > 1 ) then
469  if ( self%orig(aparticles%incidentEdges(1,self%orig(edgeindex))) == self%orig(edgeindex) ) then
470  edge1vec = edgevector( self, aparticles%incidentEdges(1,self%orig(edgeindex)), aparticles )
471  elseif ( self%dest(aparticles%incidentEdges(1,self%orig(edgeindex))) == self%orig(edgeindex) ) then
472  edge1vec = -edgevector(self, aparticles%incidentEdges(1,self%orig(edgeindex)), aparticles)
473  else

475  return
476  endif
477  newedgevec = edgevector( self, edgeindex, aparticles)
478  edge1vec = edge1vec/sqrt(sum( edge1vec*edge1vec))
479  newedgevec = newedgevec/sqrt(sum( newedgevec*newedgevec))
480 
481  cp = crossproduct(edge1vec, newedgevec)
482  edgeangleatorig = atan2( sqrt(sum(cp*cp)), sum(edge1vec*newedgevec))
483  endif

486  endif
