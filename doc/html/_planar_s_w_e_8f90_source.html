220  real(kint) :: denom
221 
222  call setfieldtozero(velocity)
223  velocity%N = self%mesh%particles%N

226  do j = 1, self%mesh%particles%N
227  if ( self%mesh%particles%isActive(j) ) then
228  denom = 2.0_kreal * pi * ( (x(i) - x(j))**2 + (y(i) - y(j))**2)
229  velocity%xComp(i) = velocity%xComp(i) + (( y(i) - y(j) ) * relvort(j) + (x(i) - x(j)) * divergence(j)) * &
230  area(j) / denom
231  velocity%yComp(i) = velocity%yComp(i) + (-(x(i) - x(j) ) * relvort(j) + (y(i) - y(j)) * divergence(j)) * &
232  area(j) / denom
233  endif
234  enddo
235  enddo
236 
237  do i = 0, numprocs - 1
238  call mpi_bcast(velocity%xComp(self%mpiParticles%indexStart(i):self%mpiParticles%indexEnd(i)), &
239  self%mpiParticles%messageLength(i), mpi_double_precision, i, mpi_comm_world, mpierrcode)
240  call mpi_bcast(velocity%yComp(self%mpiParticles%indexStart(i):self%mpiParticles%indexEnd(i)), &
